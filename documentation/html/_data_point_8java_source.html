129  double gradient() {
130  //Trying to calculate the gradient with the nextDataPoint
131  if((this.prevDataPoint==null & this.nextDataPoint!=null) || (this.prevDataPoint!=null & this.nextDataPoint!=null)) {
132  return (this.nextDataPoint.x - this.x)/(this.nextDataPoint.t - this.t);
133  }
134  //If there isn't any nextDataPoint, we will use the prevDataPoint instead.
135  else if(this.prevDataPoint!=null & this.nextDataPoint==null){
136  return (this.x - this.prevDataPoint.x)/(this.t - this.prevDataPoint.t);
137  }
138  //If there isn't any prevDataPoint, we will return 0 (because only one DataPoint hasn't any gradient).
139  else {
140  return 0;
141  }
142  }
